## @brief nextra-content — blog, wiki, polls, gallery.
cmake_minimum_required(VERSION 3.20)
project(nextra-content VERSION 0.1.0 LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

include(
    "${CMAKE_CURRENT_SOURCE_DIR}/../../cmake/service.cmake"
)

nextra_service(
    NAME nextra-content
    DIRS
        blog/backend
        blog/controllers
        wiki/backend
        wiki/controllers
        polls/backend
        polls/controllers
        gallery/backend
        gallery/controllers
        job-queue/backend
        search/events
)
